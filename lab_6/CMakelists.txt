cmake_minimum_required(VERSION 3.10)
project(lab_6)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Werror")
elseif(MSVC)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /W4")
endif()


include(FetchContent)
FetchContent_Declare(
  googletest
  GIT_REPOSITORY https://github.com/google/googletest.git
  GIT_TAG v1.15.0
  TLS_VERIFY false
)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)


add_library(${CMAKE_PROJECT_NAME}_lib 
    src/npc.cpp
    src/npc_factory.cpp
    src/visitor.cpp
    src/observer.cpp
    src/dungeon.cpp
)

target_include_directories(${CMAKE_PROJECT_NAME}_lib PUBLIC include)


if(WIN32)
    target_compile_definitions(${CMAKE_PROJECT_NAME}_lib PRIVATE _CRT_SECURE_NO_WARNINGS)
endif()


add_executable(${CMAKE_PROJECT_NAME} src/main.cpp)
target_link_libraries(${CMAKE_PROJECT_NAME} PRIVATE ${CMAKE_PROJECT_NAME}_lib)


enable_testing()

add_executable(tests 
    tests/test_main.cpp
    tests/test_npc.cpp
    tests/test_factory.cpp
    tests/test_dungeon.cpp
    tests/test_combat.cpp
    tests/test_observer.cpp
)

target_link_libraries(tests ${CMAKE_PROJECT_NAME}_lib gtest_main)
target_include_directories(tests PRIVATE include)

add_test(NAME DungeonEditorTests COMMAND tests)
